import type { MetadataRoute } from "next";

const SITE = process.env.NEXT_PUBLIC_SITE_URL ?? "https://timeproofs.io";

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: "*",
      allow: "/",
      disallow: [
        "/api/",
        "/internal/",
        "/drafts/",
        "/admin/",
        "/_next/",
        "/assets/private/",
        "/tmp/",
        "/*?preview=*",
        "/*?debug=*",
      ],
    },
    sitemap: `${SITE}/sitemap.xml`,
  };
}
